        this.nazwa = nazwa;

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;

        TurystaEntity that = (TurystaEntity) o;

        if (nazwa != null ? !nazwa.equals(that.nazwa) : that.nazwa != null) return false;
